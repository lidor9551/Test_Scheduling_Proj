  cmake -S . -B build

  ./build/exam_scheduler --courses sample_input/courses.txt --periods sample_input/periods.txt --selected-programs sample_input/selected_programs.txt --output output.txt

  ./build/exam_scheduler --courses test_inputs/no_solution_courses.txt --periods test_inputs/no_solution_periods.txt --selected-programs test_inputs/no_solution_selected_programs.txt --output no_solution_output.txt

Run elective-overlap test:
  ./build/exam_scheduler --courses test_inputs/elective_overlap_courses.txt --periods test_inputs/elective_overlap_periods.txt --selected-programs test_inputs/elective_overlap_selected_programs.txt --output elective_output.txt --combined-limit 20
